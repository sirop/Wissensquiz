74 void XmlSyntaxHighlighter::highlightBlock(const QString &text)
75 {
76  foreach (const HighlightingRule &rule, highlightingRules) {
77  QRegExp expression(rule.pattern);
78  int index = text.indexOf(expression);
79  while (index >= 0) {
80  int length = expression.matchedLength();
81  setFormat(index, length, rule.format);
82  index = text.indexOf(expression, index + length);
83  }
84  }
85  setCurrentBlockState(0);
86 
87  int startIndex = 0;
88  if (previousBlockState() != 1)
89  startIndex = text.indexOf(commentStartExpression);
90 
91  while (startIndex >= 0) {
92  int endIndex = text.indexOf(commentEndExpression, startIndex);
93  int commentLength;
94  if (endIndex == -1) {
95  setCurrentBlockState(1);
96  commentLength = text.length() - startIndex;
97  } else {
98  commentLength = endIndex - startIndex
99  + commentEndExpression.matchedLength();
100  }
101  setFormat(startIndex, commentLength, commentFormat);
102  startIndex = text.indexOf(commentStartExpression,
103  startIndex + commentLength);
104  }
105 }
